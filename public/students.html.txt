fetch("/students")
  .then(res => res.json())
  .then(data => {
    const table = document.getElementById("studentTable");
    data.forEach(s => {
      table.innerHTML += `
        <tr>
          <td>${s.name}</td>
          <td>${s.email}</td>
          <td>${s.course}</td>
        </tr>
      `;
    });
  });
